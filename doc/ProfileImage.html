def self.minimun_requirement?(user_id, profile_id)
  profile_image = ProfileImage.where(user_id: user_id, profile_id: profile_id).first
  if profile_image.present?
    if profile_image.image.present?
      return true
    else
      return false
    end
  else
    return false
  end
end
